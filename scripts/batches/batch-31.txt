Export this class:

class BigQueryConnector {
  constructor(private readonly supabaseClient: SupabaseClient) {}

  // Retrieve and decrypt a connection's service account credentials for an org.
  // Throws if connection not found or not active.
  async getConnection(orgId: string, connectionId: string): Promise<BigQueryConnectionRow>

  // Decrypt service_account_key_encrypted and return parsed JSON object.
  // Uses BIGQUERY_ENCRYPTION_KEY env var. Throws if env var missing.
  private decryptServiceAccountKey(encrypted: string): Record<string, unknown>

  // Encrypt a service account JSON string for storage.
  // Static so it can be called before class instantiation.
  static encryptServiceAccountKey(plaintext: string): string

  // List all datasets in the GCP project for a given connection.
  async listDatasets(orgId: string, connectionId: string): Promise<BigQueryDataset[]>

  // List all tables in a specific dataset.
  async listTables(orgId: string, connectionId: string, datasetId: string): Promise<BigQueryTable[]>

  // Get schema and a bounded sample (max 20 rows) for a specific table.
  async getTableSchema(orgId: string, connectionId: string, datasetId: string, tableId: string): Promise<BigQuerySchemaResult>

  // Execute a SQL query. maxRows caps the result at 10,000 rows (hard ceiling — never exceed this).
  // Returns rows, schema, rowCount, executionTimeMs, and jobId.
  // Throws descriptive errors for auth failure, invalid SQL, quota exceeded, or timeout.
  async executeQuery(orgId: string, connectionId: string, sql: string, maxRows?: number): Promise<BigQueryQueryResult>

  // Test a connection by running SELECT 1. Returns true if successful, throws with message if not.
  async testConnection(orgId: string, connectionId: string): Promise<boolean>
}
